- 每次冒泡时，参与的元素应该依次减少
- ```
  function sort(arr){
  	for(let i = 0; i < arr.length ;i++){
      	for(let j = 0;j < arr.length - 1 - i;j++){
          	if(arr[j] > arr[j + 1]){
              	const temp = arr[j];
                  arr[j] = arr[j+ 1];
                  arr[j+1] = temp
              }
          }
      }
      
      return arr;
  }
  ```
