  function throttle(func,wait){
  	let preTime = 0;
      return function(){
      	let newTime = +new Date(); //+号操作是将元素转化成NumBer类
          let context = this;
          let args = arguments;
          
          if(newTime - preTime > wait){
          	func(context,args);
              preTime = newTime;
          }
      }
  }
